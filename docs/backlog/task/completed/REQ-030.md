### 2.2 既存の対象コード
```typescript
// src/store/visionStore.ts
export const useVisionStore = create<VisionState>((set, get) => ({
  status: 'idle',
  currentJobId: null,
  videoMeta: null,
  progress: 0,
  isAnalyzing: false,
  talismans: [],
  profiles: [],
  error: null,
  analysisConfig: { scroll_pace_seconds: 0.3, stillness_threshold: 0.01 }, // 初期値追加

  startAnalysis: async (profileId: string) => {
    const { currentJobId } = get();
    if (!currentJobId) return;
    try {
      set({ status: 'processing', isAnalyzing: true, error: null });
      // 修正: analysis_config を含める
      await startAnalysis(currentJobId, { 
        profile_id: profileId,
        analysis_config: get().analysisConfig 
      });
      get().listenToEvents(currentJobId);
    } catch (err: any) {
      set({ status: 'failed', error: '解析の開始に失敗しました' });
    }
  },
  
  updateAnalysisConfig: (config: Partial<AnalysisConfig>) => {
    set((state) => ({
      analysisConfig: { ...state.analysisConfig, ...config }
    }));
  },
  // ...
}));
```
